import random  # computer selects randomly-generated options for each round/game

# GLOBAL VARIABLES

# GAME STATISTICS
'''These counters track the user's performance across the game'''
total_games_played = 0
total_wins = 0
total_losses = 0

# GAME DICTIONARY
'''This dictionary stores the option choices for display'''
display_choice_dict = {"r": "Rock",
                       "p": "Paper",
                       "s": "Scissors"}

# USER'S NAME
name = ""
name = welcome(name)

# NAME VALIDATION
def ask_name():
    print("Welcome to a game of Rock, Paper, Scissors!") # intro message
    ''' Prompts user to enter a valid name'''
    attempts = 5 # the user has 5 attempts before the game forces an end
    
    while attempts > 0:  # as long as attempts is greater than 0, keep running
        name_input = input("Please Enter Your Name: ")
        
        if name_input.isalpha() and len(name_input) >= 3: #a valid name must contain only alphabetical characters
            # and a valid name must be at least 3 characters long
            return name_input
        else:
            attempts -= 1
            print("Invalid Input")
            print(">> Your name must contain only letters and be at least 3 characters long")
            print(">> Attempts remaining: %i\n" % attempts)
            
    return ""

# WELCOME MESSAGE
def welcome(username=""):
    ''' Returns the validated username, calls ask_name() if no username is given '''
    ''' Displays instructions of the game on screen for user '''
    
    if len(username) == 0:
        username = ask_name()
        
    if len(username) == 0:
        return ""
        
    print("Hello %s." % username)
    print("Let's start playing!\n")
    print("Instructions:")
    print("- Enter 'r' for Rock")
    print("- Enter 'p' for Paper")
    print("- Enter 's' for Scissors")
    print("- Enter 'e' to exit the game")
    print()
    
    return username

# USER SELECTION
def get_user_selection():
    ''' Keeps asking user to input selection of options until a valid option is entered'''
    ''' If user enters 'e', function signals game to end'''
    
    user_options = ["r", "p", "s", "e"] 
    
    while True:
        selection = input("Enter your choice (r/p/s/e):").lower()

        if selection in user_options:
            return selection
            
        else:
            print("Invalid selection.")
            print("Please enter 'r' for Rock, 'p' for Paper, 's' for Scissors, or 'e' to Exit the game.\n")

# COMPUTER SELECTION
def get_computer_selection():
    ''' Randomly selects Rock, Paper, or Scissors'''
    computer_options = ["r", "p", "s"]
    return random.choice(computer_options)

# DISPLAY CHOICE
def display_choice(selection):
    ''' Converts the short one letter option into a full word for display'''
    return display_choice_dict[selection]

# DETERMINE WINNER
def determine_winner(user_choice, computer_choice):
    '''Returns result type and the displays it as a message to the user'''
    print("\nYou chose: %s. Computer chose: %s." % (display_choice(user_choice), display_choice(computer_choice)))

    if user_choice == computer_choice:
        return ["tie", "It's a tie!"]

    elif user_choice == "r":
        if computer_choice == "s":
            return ["win", "Rock smashes scissors! You win!"]
        else:
            return ["loss", "Paper covers rock. You lose."]

    elif user_choice == "p":
        if computer_choice == "r":
            return ["win", "Paper covers rock! You win!"]
        else:
            return ["loss", "Scissors shred paper. You lose."]

    elif user_choice == "s":
        if computer_choice == "p":
            return ["win", "Scissors shred paper! You win!"]
        else:
            return ["loss", "Rock smashes scissors. You lose."]

# ASK TO PLAY AGAIN
def ask_play_again():
    '''Asks player if they want to play again, repeats until user enters either 'y' or 'n'''
    while True:
        answer = input("Would you like to play again? (y/n): ").lower()
        
        if answer in ["y", "n"]:
            return answer
        else:
            print("Invalid input")
            print("Please enter 'y' to continue, or 'n' to stop.\n")

# GAME FLOW
if len(name) == 0: # validate name
    print("Game ended due to invalid name input.")
    
else:
    total_wins = 0
    total_losses = 0

    while True:

        user_selection = get_user_selection()
        
        if user_selection == "e": #game exits if user selects 'e'
            break

        computer_selection = get_computer_selection()

        result = determine_winner(user_selection, computer_selection)
        result_message = result[1]
        result = result[0]
        
        print(result_message)

        total_games_played += 1

        if result == "win":
            total_wins += 1
            print("\n1 Point has been added to your Wins!\n")
            
        elif result == "loss":
            total_losses += 1
            print("\n1 Point has been added to your Losses...\n")
            
        else:
            print("\nNo Points have been added\n")

        print("------ CURRENT SCORE SUMMARY ------")
        print("Wins:", total_wins)
        print("Losses:", total_losses)
        print("Total Games Played:", total_games_played)
        print()

        again = ask_play_again()

        if again == "y":
            continue
            
        else:
            break
            
    print("\nThank you for playing.")
    print("------ FINAL SCORE SUMMARY ------")
    print("Wins:", total_wins)
    print("Losses:", total_losses)
    print("Total Games Played:", total_games_played)
    print("GAME ENDED")
